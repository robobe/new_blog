    from scapy.all import rdpcap, IP, UDP, send
import time
PCAP = "mavlink.pcap"
pkts = rdpcap(PCAP)

t0 = None
start = None

for p in pkts:
    if IP in p and UDP in p:
        if t0 is None:
            t0 = float(p.time)
            start = time.time()

        pkt_time = float(p.time)
        delay = (pkt_time - t0) - (time.time() - start)

        if delay > 0:
            time.sleep(delay)

        del p[IP].len
        del p[IP].chksum
        del p[UDP].len
        del p[UDP].chksum

        send(p[IP], verbose=False)
